        while(n > 0){
            digit.push_back(n % 10);
            n /= 10;
        }

        for(int i=digit.size()-1; i>=0; i--){
            if(digit[i] != 0){
                sum += digit[i];
                num = num * 10 + digit[i];
            }
        }
        
        return num* sum;
    }
};
