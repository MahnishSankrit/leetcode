                    high = mid - 1;
                }else{
                if(nums[mid] != nums[mid + 1]){
                    low = mid + 2;
                }
            }else {
                if(nums[mid] != nums[mid -1] ){
                    high = mid - 1; 
                }else{
                    low = mid + 1;
                }
            }
        }
        return nums[low];
    }
