class Solution {
public:
    bool rotateString(string s, string goal) {
        int n=s.length();
        for(int i=0; i<n; i++){
            if(s != goal){
                rotate(s.begin(), s.begin() + 1, s.end());
            }else{
                return true;
            }
        }

        return false;
    }
};
