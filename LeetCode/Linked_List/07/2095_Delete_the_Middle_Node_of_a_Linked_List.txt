
        ListNode *slow = head;
        ListNode *fast = head;

        ListNode *prev = nullptr;
        while(fast != nullptr && fast->next != nullptr){
            prev = slow;
            slow = slow->next;
            fast = fast->next->next;
        }

        prev->next = slow->next;

        slow->next = nullptr;
        if(head->next == nullptr) return nullptr;
        delete slow;
        return head;
    }
};
